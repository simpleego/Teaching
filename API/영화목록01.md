## 3. JavaScript 파일 (script.js)

```javascript
// DOM 요소
const apiKeyInput = document.getElementById('api-key');
const toggleKeyBtn = document.getElementById('toggle-key');
const loadMoviesBtn = document.getElementById('load-movies');
const clearKeyBtn = document.getElementById('clear-key');
const targetDateInput = document.getElementById('target-date');
const moviesContainer = document.getElementById('movies-container');
const movieCount = document.getElementById('movie-count');
const loadingSection = document.getElementById('loading');
const detailSection = document.getElementById('detail-section');
const detailContent = document.getElementById('detail-content');
const closeDetailBtn = document.getElementById('close-detail');
const notification = document.getElementById('notification');

// API 기본 URL
const KOBIS_BASE_URL = 'http://www.kobis.or.kr/kobisopenapi/webservice/rest';

// 오늘 날짜 설정 (기본값: 어제 날짜)
function setDefaultDate() {
    const today = new Date();
    const yesterday = new Date(today);
    yesterday.setDate(yesterday.getDate() - 1);
    
    const formattedDate = yesterday.toISOString().split('T')[0];
    targetDateInput.value = formattedDate;
    targetDateInput.max = formattedDate; // 오늘 이후 날짜 선택 불가
}

// API 키 표시/숨기기 토글
toggleKeyBtn.addEventListener('click', () => {
    const type = apiKeyInput.getAttribute('type');
    if (type === 'password') {
        apiKeyInput.setAttribute('type', 'text');
        toggleKeyBtn.innerHTML = '<i class="fas fa-eye-slash"></i>';
    } else {
        apiKeyInput.setAttribute('type', 'password');
        toggleKeyBtn.innerHTML = '<i class="fas fa-eye"></i>';
    }
});

// API 키 지우기
clearKeyBtn.addEventListener('click', () => {
    apiKeyInput.value = '';
    showNotification('API 키가 삭제되었습니다.', 'info');
});

// 날짜 형식 변환 (YYYY-MM-DD -> YYYYMMDD)
function formatDateForAPI(dateString) {
    return dateString.replace(/-/g, '');
}

// 현재 상영중인 영화 목록 가져오기
async function fetchNowShowingMovies() {
    const apiKey = apiKeyInput.value.trim();
    const targetDate = targetDateInput.value;
    
    // 유효성 검사
    if (!apiKey) {
        showNotification('API 키를 입력해주세요.', 'error');
        apiKeyInput.focus();
        return;
    }
    
    if (!targetDate) {
        showNotification('조회 기준일을 선택해주세요.', 'error');
        targetDateInput.focus();
        return;
    }
    
    // 로딩 표시
    loadingSection.style.display = 'block';
    moviesContainer.innerHTML = '';
    
    try {
        const formattedDate = formatDateForAPI(targetDate);
        const url = `${KOBIS_BASE_URL}/boxoffice/searchDailyBoxOfficeList.json`;
        
        const params = new URLSearchParams({
            key: apiKey,
            targetDt: formattedDate,
            itemPerPage: '10',
            multiMovieYn: 'Y',
            repNationCd: '',
            wideAreaCd: ''
        });
        
        const response = await fetch(`${url}?${params}`);
        
        if (!response.ok) {
            throw new Error(`API 요청 실패: ${response.status}`);
        }
        
        const data = await response.json();
        
        // API 응답 확인
        if (data.faultInfo) {
            throw new Error(data.faultInfo.message || 'API 키가 유효하지 않거나 요청에 문제가 있습니다.');
        }
        
        const movieList = data.boxOfficeResult?.dailyBoxOfficeList || [];
        
        if (movieList.length === 0) {
            showNotification('선택한 날짜에 상영중인 영화 정보가 없습니다.', 'info');
            displayNoMovies();
        } else {
            displayMovies(movieList);
            showNotification(`${movieList.length}개의 영화 정보를 불러왔습니다.`, 'success');
        }
        
    } catch (error) {
        console.error('영화 정보 조회 중 오류 발생:', error);
        showNotification(`오류 발생: ${error.message}`, 'error');
        displayNoMovies();
    } finally {
        loadingSection.style.display = 'none';
    }
}

// 영화 목록 표시
function displayMovies(movies) {
    moviesContainer.innerHTML = '';
    movieCount.textContent = `총 ${movies.length}개의 영화`;
    
    movies.forEach((movie, index) => {
        const movieCard = createMovieCard(movie, index + 1);
        moviesContainer.appendChild(movieCard);
    });
}

// 영화 카드 생성
function createMovieCard(movie, rank) {
    const card = document.createElement('div');
    card.className = 'movie-card';
    card.dataset.movieCd = movie.movieCd;
    
    // 개봉일 포맷팅
    const openDate = movie.openDt;
    const formattedOpenDate = openDate && openDate.length === 8 
        ? `${openDate.substring(0, 4)}-${openDate.substring(4, 6)}-${openDate.substring(6, 8)}`
        : '정보 없음';
    
    // 관객수 포맷팅 (천단위 구분)
    const formatNumber = (num) => {
        if (!num || isNaN(num)) return '0';
        return parseInt(num).toLocaleString('ko-KR');
    };
    
    card.innerHTML = `
        <div class="movie-rank">${rank}</div>
        <div class="movie-poster">
            <i class="fas fa-film"></i>
        </div>
        <div class="movie-info">
            <h3 class="movie-title">${movie.movieNm}</h3>
            <div class="movie-meta">
                <div class="meta-item">
                    <span class="meta-label">개봉일</span>
                    <span class="meta-value">${formattedOpenDate}</span>
                </div>
                <div class="meta-item">
                    <span class="meta-label">영화 코드</span>
                    <span class="meta-value">${movie.movieCd}</span>
                </div>
                <div class="meta-item">
                    <span class="meta-label">순위 변화</span>
                    <span class="meta-value ${parseInt(movie.rankInten) > 0 ? 'rank-up' : parseInt(movie.rankInten) < 0 ? 'rank-down' : ''}">
                        ${parseInt(movie.rankInten) > 0 ? '▲' : parseInt(movie.rankInten) < 0 ? '▼' : '─'} ${Math.abs(parseInt(movie.rankInten))}
                    </span>
                </div>
            </div>
            <div class="movie-audience">
                <div class="meta-label">일일 관객수</div>
                <div class="audience-count">${formatNumber(movie.audiCnt)}명</div>
                <div class="meta-label">누적 관객수: ${formatNumber(movie.audiAcc)}명</div>
            </div>
        </div>
    `;
    
    // 클릭 이벤트 - 상세 정보 보기
    card.addEventListener('click', () => {
        fetchMovieDetail(movie.movieCd);
    });
    
    return card;
}

// 영화 상세 정보 가져오기
async function fetchMovieDetail(movieCd) {
    const apiKey = apiKeyInput.value.trim();
    
    if (!apiKey) {
        showNotification('API 키를 입력해주세요.', 'error');
        return;
    }
    
    try {
        const url = `${KOBIS_BASE_URL}/movie/searchMovieInfo.json`;
        const params = new URLSearchParams({
            key: apiKey,
            movieCd: movieCd
        });
        
        const response = await fetch(`${url}?${params}`);
        
        if (!response.ok) {
            throw new Error(`상세 정보 요청 실패: ${response.status}`);
        }
        
        const data = await response.json();
        
        if (data.faultInfo) {
            throw new Error(data.faultInfo.message || '영화 상세 정보를 가져오는데 실패했습니다.');
        }
        
        const movieInfo = data.movieInfoResult?.movieInfo || {};
        displayMovieDetail(movieInfo);
        
    } catch (error) {
        console.error('영화 상세 정보 조회 중 오류 발생:', error);
        showNotification(`상세 정보 오류: ${error.message}`, 'error');
    }
}

function showNotification(message, type = 'info') {
    notification.textContent = message;
    notification.className = 'notification';
    
    // 타입에 따른 색상 설정
    if (type === 'error') {
        notification.style.background = 'rgba(255, 107, 107, 0.9)';
    } else if (type === 'success') {
        notification.style.background = 'rgba(76, 209, 96, 0.9)';
    } else {
        notification.style.background = 'rgba(77, 150, 255, 0.9)';
    }
    
    notification.classList.add('show');
    
    // 3초 후 알림 숨기기

        notification.classList.remove('show');
    }, 3000);
}
```
