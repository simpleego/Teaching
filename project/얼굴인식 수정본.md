    <script src="https://cdn.jsdelivr.net/npm/@tensorflow/tfjs@3.18.0/dist/tf.min.js"></script>
    <script src="https://cdn.jsdelivr.net/npm/face-api.js@0.22.2/dist/face-api.min.js"></script>
    
    <script>
        // DOM 요소
        const video = document.getElementById('video');
        const overlay = document.getElementById('overlay');
        const loadModelsBtn = document.getElementById('loadModels');
        const startCameraBtn = document.getElementById('startCamera');
        const stopCameraBtn = document.getElementById('stopCamera');
        const registerFaceBtn = document.getElementById('registerFace');
        const checkAttendanceBtn = document.getElementById('checkAttendance');
        const studentNameInput = document.getElementById('studentName');
        const messageDiv = document.getElementById('message');
        const attendanceList = document.getElementById('attendanceList');
        
        // 전역 변수
        let stream = null;
        let faceDetectionInterval = null;
        let registeredFaces = [];
        let attendanceRecords = [];
        let modelsLoaded = false;
        
        // 얼굴 인식 모델 로드
        async function loadModels() {
            try {
                showMessage('<div class="loading"></div>얼굴 인식 모델을 로드 중입니다...', 'info');
                loadModelsBtn.disabled = true;
                loadModelsBtn.innerHTML = '모델 로드 중...';
                
                // CDN에서 직접 모델 로드 (로컬 파일 필요 없음)
                await faceapi.nets.tinyFaceDetector.loadFromUri('https://cdn.jsdelivr.net/npm/face-api.js@0.22.2/weights/');
                await faceapi.nets.faceLandmark68Net.loadFromUri('https://cdn.jsdelivr.net/npm/face-api.js@0.22.2/weights/');
                await faceapi.nets.faceRecognitionNet.loadFromUri('https://cdn.jsdelivr.net/npm/face-api.js@0.22.2/weights/');
                await faceapi.nets.faceExpressionNet.loadFromUri('https://cdn.jsdelivr.net/npm/face-api.js@0.22.2/weights/');
                
                modelsLoaded = true;
                startCameraBtn.disabled = false;
                loadModelsBtn.style.display = 'none';
                
                showMessage('얼굴 인식 모델 로드 완료! 이제 카메라를 시작할 수 있습니다.', 'success');
            } catch (error) {
                console.error('모델 로드 중 오류 발생:', error);
                showMessage('모델 로드 중 오류가 발생했습니다. 인터넷 연결을 확인하세요.', 'error');
                loadModelsBtn.disabled = false;
                loadModelsBtn.innerHTML = '모델 재로드';
            }
        }
        
        // 카메라 시작
        async function startCamera() {
            if (!modelsLoaded) {
                showMessage('먼저 모델을 로드해주세요.', 'error');
                return;
            }
            
            try {
                stream = await navigator.mediaDevices.getUserMedia({ 
                    video: { width: 600, height: 450 } 
                });
                video.srcObject = stream;
                
                // 비디오 크기 설정
                video.width = 600;
                video.height = 450;
                overlay.width = 600;
                overlay.height = 450;
                
                startCameraBtn.disabled = true;
                stopCameraBtn.disabled = false;
                registerFaceBtn.disabled = false;
                checkAttendanceBtn.disabled = false;
                
                showMessage('카메라가 시작되었습니다.', 'success');
                
                // 얼굴 감지 시작
                startFaceDetection();
            } catch (error) {
                console.error('카메라 시작 중 오류 발생:', error);
                showMessage('카메라 시작 중 오류가 발생했습니다. 권한을 확인하세요.', 'error');
            }
        }
        
        // 카메라 중지
        function stopCamera() {
            if (stream) {
                stream.getTracks().forEach(track => track.stop());
                stream = null;
            }
            
            if (faceDetectionInterval) {
                clearInterval(faceDetectionInterval);
                faceDetectionInterval = null;
            }
            
            // 비디오 요소 초기화
            video.srcObject = null;
            
            // 오버레이 캔버스 초기화
            const ctx = overlay.getContext('2d');
            ctx.clearRect(0, 0, overlay.width, overlay.height);
            
            startCameraBtn.disabled = false;
            stopCameraBtn.disabled = true;
            registerFaceBtn.disabled = true;
            checkAttendanceBtn.disabled = true;
            
            showMessage('카메라가 중지되었습니다.', 'info');
        }
        
        // 얼굴 감지 시작
        function startFaceDetection() {
            if (faceDetectionInterval) {
                clearInterval(faceDetectionInterval);
            }
            
            faceDetectionInterval = setInterval(async () => {
                if (video.paused || video.ended) return;
                
                try {
                    // 얼굴 감지
                    const detections = await faceapi
                        .detectAllFaces(video, new faceapi.TinyFaceDetectorOptions())
                        .withFaceLandmarks()
                        .withFaceDescriptors();
                    
                    // 오버레이 그리기
                    const displaySize = { width: video.width, height: video.height };
                    faceapi.matchDimensions(overlay, displaySize);
                    
                    const resizedDetections = faceapi.resizeResults(detections, displaySize);
                    const ctx = overlay.getContext('2d');
                    ctx.clearRect(0, 0, overlay.width, overlay.height);
                    
                    // 얼굴 경계 상자 그리기
                    resizedDetections.forEach(detection => {
                        const box = detection.detection.box;
                        ctx.strokeStyle = '#00ff00';
                        ctx.lineWidth = 2;
                        ctx.strokeRect(box.x, box.y, box.width, box.height);
                        
                        // 얼굴 인식 여부 표시
                        const recognizedFace = recognizeFace(detection.descriptor);
                        if (recognizedFace) {
                            ctx.fillStyle = '#00ff00';
                            ctx.font = '16px Arial';
                            ctx.fillText(recognizedFace.name, box.x, box.y - 10);
                        } else {
                            ctx.fillStyle = '#ff0000';
                            ctx.font = '16px Arial';
                            ctx.fillText('알 수 없는 얼굴', box.x, box.y - 10);
                        }
                    });
                } catch (error) {
                    console.error('얼굴 감지 중 오류 발생:', error);
                }
            }, 300); // 300ms 간격으로 감지 (성능 고려)
        }
        
        // 얼굴 등록
        async function registerFace() {
            const studentName = studentNameInput.value.trim();
            if (!studentName) {
                showMessage('학생 이름을 입력해주세요.', 'error');
                return;
            }
            
            try {
                // 얼굴 감지
                const detections = await faceapi
                    .detectAllFaces(video, new faceapi.TinyFaceDetectorOptions())
                    .withFaceLandmarks()
                    .withFaceDescriptors();
                
                if (detections.length === 0) {
                    showMessage('얼굴이 감지되지 않았습니다. 카메라 앞으로 이동해주세요.', 'error');
                    return;
                }
                
                if (detections.length > 1) {
                    showMessage('여러 얼굴이 감지되었습니다. 한 명만 카메라 앞에 서주세요.', 'error');
                    return;
                }
                
                // 얼굴 디스크립터 저장
                const faceDescriptor = detections[0].descriptor;
                
                // 이미 등록된 이름인지 확인
                if (registeredFaces.some(face => face.name === studentName)) {
                    showMessage(`이미 등록된 이름입니다: ${studentName}`, 'error');
                    return;
                }
                
                // 얼굴 등록
                registeredFaces.push({
                    name: studentName,
                    descriptor: Array.from(faceDescriptor) // Float32Array를 일반 배열로 변환
                });
                
                // 출석 기록에 추가 (기본값: 결석)
                const today = new Date().toISOString().split('T')[0];
                attendanceRecords.push({
                    name: studentName,
                    date: today,
                    status: 'absent'
                });
                
                showMessage(`얼굴 등록 완료: ${studentName}`, 'success');
                studentNameInput.value = '';
                
                // 출석 목록 업데이트
                updateAttendanceList();
                
                // 로컬 스토리지에 저장
                saveToLocalStorage();
            } catch (error) {
                console.error('얼굴 등록 중 오류 발생:', error);
                showMessage('얼굴 등록 중 오류가 발생했습니다.', 'error');
            }
        }
        
        // 얼굴 인식
        function recognizeFace(descriptor) {
            if (registeredFaces.length === 0) return null;
            
            let bestMatch = null;
            let bestDistance = Number.MAX_VALUE;
            
            // 등록된 얼굴과 비교
            registeredFaces.forEach(registeredFace => {
                // 일반 배열을 Float32Array로 변환
                const registeredDescriptor = new Float32Array(registeredFace.descriptor);
                const distance = faceapi.euclideanDistance(descriptor, registeredDescriptor);
                
                // 임계값 설정 (0.6은 일반적으로 사용되는 값)
                if (distance < 0.6 && distance < bestDistance) {
                    bestDistance = distance;
                    bestMatch = registeredFace;
                }
            });
            
            return bestMatch;
        }
        
        // 출석 체크
        async function checkAttendance() {
            try {
                // 얼굴 감지
                const detections = await faceapi
                    .detectAllFaces(video, new faceapi.TinyFaceDetectorOptions())
                    .withFaceLandmarks()
                    .withFaceDescriptors();
                
                if (detections.length === 0) {
                    showMessage('얼굴이 감지되지 않았습니다. 카메라 앞으로 이동해주세요.', 'error');
                    return;
                }
                
                // 각 감지된 얼굴에 대해 출석 체크
                let attendanceChecked = false;
                const today = new Date().toISOString().split('T')[0];
                
                detections.forEach(detection => {
                    const recognizedFace = recognizeFace(detection.descriptor);
                    if (recognizedFace) {
                        // 출석 기록 업데이트
                        const recordIndex = attendanceRecords.findIndex(
                            record => record.name === recognizedFace.name && record.date === today
                        );
                        
                        if (recordIndex !== -1) {
                            attendanceRecords[recordIndex].status = 'present';
                            attendanceChecked = true;
                            showMessage(`출석 체크 완료: ${recognizedFace.name}`, 'success');
                        }
                    }
                });
                
                if (!attendanceChecked) {
                    showMessage('등록된 얼굴이 감지되지 않았습니다.', 'error');
                } else {
                    // 출석 목록 업데이트
                    updateAttendanceList();
                    // 로컬 스토리지에 저장
                    saveToLocalStorage();
                }
            } catch (error) {
                console.error('출석 체크 중 오류 발생:', error);
                showMessage('출석 체크 중 오류가 발생했습니다.', 'error');
            }
        }
        
        // 출석 목록 업데이트
        function updateAttendanceList() {
            attendanceList.innerHTML = '';
            
            if (attendanceRecords.length === 0) {
                attendanceList.innerHTML = '<p>등록된 학생이 없습니다.</p>';
                return;
            }
            
            // 오늘 날짜 기준으로 필터링
            const today = new Date().toISOString().split('T')[0];
            const todayRecords = attendanceRecords.filter(record => record.date === today);
            
            if (todayRecords.length === 0) {
                attendanceList.innerHTML = '<p>오늘의 출석 기록이 없습니다.</p>';
                return;
            }
            
            todayRecords.forEach(record => {
                const studentItem = document.createElement('div');
                studentItem.className = 'student-item';
                
                const studentInfo = document.createElement('div');
                studentInfo.className = 'student-info';
                
                const avatar = document.createElement('div');
                avatar.className = 'student-avatar';
                avatar.textContent = record.name.charAt(0).toUpperCase();
                
                const name = document.createElement('span');
                name.textContent = record.name;
                
                studentInfo.appendChild(avatar);
                studentInfo.appendChild(name);
                
                const status = document.createElement('div');
                status.className = `status ${record.status}`;
                status.textContent = record.status === 'present' ? '출석' : '결석';
                
                studentItem.appendChild(studentInfo);
                studentItem.appendChild(status);
                
                attendanceList.appendChild(studentItem);
            });
        }
        
        // 로컬 스토리지에 데이터 저장
        function saveToLocalStorage() {
            const data = {
                registeredFaces: registeredFaces,
                attendanceRecords: attendanceRecords
            };
            localStorage.setItem('faceAttendanceSystem', JSON.stringify(data));
        }
        
        // 로컬 스토리지에서 데이터 로드
        function loadFromLocalStorage() {
            const data = localStorage.getItem('faceAttendanceSystem');
            if (data) {
                const parsedData = JSON.parse(data);
                registeredFaces = parsedData.registeredFaces || [];
                attendanceRecords = parsedData.attendanceRecords || [];
                updateAttendanceList();
            }
        }
        
        // 메시지 표시
        function showMessage(text, type) {
            messageDiv.innerHTML = text;
            messageDiv.className = `message ${type}`;
            
            // 5초 후 메시지 숨기기 (로딩 메시지는 제외)
            if (!text.includes('loading')) {
                setTimeout(() => {
                    messageDiv.textContent = '';
                    messageDiv.className = 'message';
                }, 5000);
            }
        }
        
        // 이벤트 리스너 등록
        loadModelsBtn.addEventListener('click', loadModels);
        startCameraBtn.addEventListener('click', startCamera);
        stopCameraBtn.addEventListener('click', stopCamera);
        registerFaceBtn.addEventListener('click', registerFace);
        checkAttendanceBtn.addEventListener('click', checkAttendance);
        
        // 페이지 로드 시 로컬 스토리지에서 데이터 로드
        window.addEventListener('load', () => {
            loadFromLocalStorage();
        });
    </script>
