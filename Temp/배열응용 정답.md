## **3. 배열의 합과 평균**
```java
import java.util.Random;

public class Q3 {
    public static void main(String[] args) {
        int[] arr = new int[10];
        Random r = new Random();
        int sum = 0;

        for (int i = 0; i < arr.length; i++) {
            arr[i] = r.nextInt(100) + 1;
            sum += arr[i];
        }

        System.out.println("합: " + sum);
        System.out.println("평균: " + (sum / (double)arr.length));
    }
}
```

## **10. 배열 정렬(버블 정렬)**
```java
import java.util.Random;

public class Q10 {
    public static void main(String[] args) {
        int[] arr = new int[10];
        Random r = new Random();

        for (int i = 0; i < arr.length; i++) arr[i] = r.nextInt(100) + 1;

        for (int i = 0; i < arr.length - 1; i++) {
            for (int j = 0; j < arr.length - 1 - i; j++) {
                if (arr[j] > arr[j + 1]) {
                    int temp = arr[j];
                    arr[j] = arr[j + 1];
                    arr[j + 1] = temp;
                }
            }
        }

        for (int n : arr) System.out.print(n + " ");
    }
}
```

## **1. 로또 번호 생성기 (중복 제거)**

```java
import java.util.Random;

public class A1 {
    public static void main(String[] args) {
        int[] lotto = new int[6];
        Random r = new Random();

        for (int i = 0; i < lotto.length; i++) {
            lotto[i] = r.nextInt(45) + 1;

            for (int j = 0; j < i; j++) {
                if (lotto[i] == lotto[j]) {
                    i--;
                    break;
                }
            }
        }

        for (int n : lotto) System.out.print(n + " ");
    }
}
```

## **2. 학생 점수 분석**
```java
import java.util.Random;

public class A2 {
    public static void main(String[] args) {
        int[] score = new int[10];
        Random r = new Random();

        int sum = 0, max = 0, min = 100;

        for (int i = 0; i < score.length; i++) {
            score[i] = r.nextInt(101);
            sum += score[i];
            if (score[i] > max) max = score[i];
            if (score[i] < min) min = score[i];
        }

        double avg = sum / 10.0;
        int above = 0;
        for (int s : score) if (s >= avg) above++;

        System.out.println("평균: " + avg);
        System.out.println("최고점: " + max);
        System.out.println("최저점: " + min);
        System.out.println("평균 이상: " + above);
    }
}
```

```java
import java.util.Random;

public class A9 {
    public static void main(String[] args) {
        int[] score = new int[20];
        int[] grade = new int[5]; // A,B,C,D,F
        Random r = new Random();

        for (int i = 0; i < score.length; i++) {
            score[i] = r.nextInt(101);

            if (score[i] >= 90) grade[0]++;
            else if (score[i] >= 80) grade[1]++;
            else if (score[i] >= 70) grade[2]++;
            else if (score[i] >= 60) grade[3]++;
            else grade[4]++;
        }

        System.out.println("A: " + grade[0]);
        System.out.println("B: " + grade[1]);
        System.out.println("C: " + grade[2]);
        System.out.println("D: " + grade[3]);
        System.out.println("F: " + grade[4]);
    }
}
```
